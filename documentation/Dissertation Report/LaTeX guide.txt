\subsection{subsection name} = New subsection under the last created section

\file{file name} = Formatting for file names within text

\url{url} = Formatting for clickable url in text

\begin{itemize}

\end{itemize}	= Bullet point list where 'text' is the contents

\keyword{keyword text} -- = Creates a bold, indented keyword with dash

\emph{text to be emphasised} = Creates italic text

\ldots = used for ellipsis (...)
